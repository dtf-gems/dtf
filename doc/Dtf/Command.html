def self.create_cmd(cmd, options)
  Dtf::Command.const_get(cmd.camelize).new(cmd, options)    
rescue NameError  
  puts "DTF has no registered command by that name."
  puts "Please see 'dtf -h' for the list of recognized commands."
end
